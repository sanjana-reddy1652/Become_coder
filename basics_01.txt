EXAMPLE 01 : SET_BITS count (c++)
#include<iostream>                 
using namespace std;
int main()
{
	int cnt=0,num;
	cin>>num;
	while(num)
	{
		if(num&1) cnt++;
		num = num>>1;
	}
	cout<<cnt;
	return 0;
}
